def load_data_from_files(filepath):
    data = []
    labels = []

    with open(filepath, 'r') as file:
        for line in file:
            parts = line.strip().split('\t')
            if len(parts) == 2:
                label, sentence = parts
                data.append(sentence)
                if label.lower() == "spam":
                    labels.append(1)
                elif label.lower() == "ham":
                    labels.append(0)

    return data, labels

def bag_of_words(data):
    vocab = set()
    for doc in data:
        vocab.update(preprocess(doc))
    vocab = list(vocab)
    return vocab

def vectorize(data, vocab):
    vectors = []
    for doc in data:
        vector = [0] * len(vocab)
        tokens = preprocess(doc)
        for token in tokens:
            if token in vocab:
                vector[vocab.index(token)] += 1
        vectors.append(vector)
    return vectors

vocab = bag_of_words(data)

from collections import defaultdict
import math

def train_bayes_classifier(training_data, training_labels, vocabulary):
    label_probabilities = defaultdict(int)
    word_likelihoods = {category: [1] * len(vocabulary) for category in set(training_labels)}
    total_word_counts = defaultdict(int)

    for i in range(len(training_data)):
        category = training_labels[i]
        label_probabilities[category] += 1
        total_word_counts[category] += len(training_data[i])
        for j in range(len(vocabulary)):
            word_likelihoods[category][j] += training_data[i][j]

    for category in label_probabilities:
        label_probabilities[category] /= len(training_labels)
        for j in range(len(vocabulary)):
            word_likelihoods[category][j] /= total_word_counts[category] + len(vocabulary)

    return label_probabilities, word_likelihoods

def classify(test_vectors, label_probabilities, word_likelihoods, vocabulary):
    predicted_labels = []
    for vector in test_vectors:
        label_scores = {}
        for label in label_probabilities:
            log_likelihood = math.log(label_probabilities[label])
            for i in range(len(vocabulary)):
                if vector[i] > 0:
                    log_likelihood += vector[i] * math.log(word_likelihoods[label][i])
            label_scores[label] = log_likelihood
        predicted_labels.append(max(label_scores, key=label_scores.get))
    return predicted_labels
